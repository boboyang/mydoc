#icmp，通常我们会把ICMP type 8 (echo request)拿掉而已，让远端主机不知道我们是否存在，也不会接受ping的回应就是了。

#!/bin/bash
icmp_type="0 3 4 11 12 14 16 18"
for typeicmp in $icmp_type
do
   iptables -A INPUT -i eth0 -p icmp --icmp-type $typeicmp -j ACCEPT
done
