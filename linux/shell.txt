#!/bin/sh

SEARCH=`date -d "2012-10-09 14:11:09" +%s`
while read DATETIME ; do
    THIS=`date -d "$DATETIME" +%s`
    if [ "$SEARCH" -le "$THIS" ] ; then
        echo $DATETIME
    fi
done
